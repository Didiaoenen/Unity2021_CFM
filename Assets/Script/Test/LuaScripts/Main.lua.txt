require("LuaPanda").start("127.0.0.1", 8818)

CS = CS or {}

require("LuaFramework/System")

function __G__TRACKBACK__(msg)
    error("----------------------------------------")
    error("LUA ERROR: " .. tostring(msg) .. "\n")
    error(debug.traceback())
    error("----------------------------------------")
end

function Main()
    class()
end

local status, msg = pcall(Main, __G__TRACKBACK__)
if not status then
    print(msg)
end
