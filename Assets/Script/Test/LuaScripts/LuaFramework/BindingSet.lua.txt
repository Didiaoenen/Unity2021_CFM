local BindingBuilder = require("BindingBuilder")

local BindingSet = class("BindingSet")

function BindingSet:cotr(context, target)
    self.super:ctor(context)
    self.target = target
end

function BindingSet:Bind()
    local builder = BindingBuilder(self.context, self.target)
    table.insert(self.builders, builder)
    return builder
end

function BindingSet:BindByTarget(target)
    local builder = BindingBuilder(self.context, target)
    table.insert(self.builders, builder)
    return builder
end

return BindingSet
