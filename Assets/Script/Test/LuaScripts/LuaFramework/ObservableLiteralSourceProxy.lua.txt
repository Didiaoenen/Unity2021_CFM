local NotifiableSourceProxyBase = require("LuaFramework/NotifiableSourceProxyBase")

local ObservableLiteralSourceProxy = class("ObservableLiteralSourceProxy", NotifiableSourceProxyBase)

function ObservableLiteralSourceProxy:ctor(source)
    self.super:ctor(source)

    if not source then
        error("")
        return
    end

    self.observableProperty = source
    self.observableProperty.ValueChanged = function ()
        self:OnValueChanged()
    end
end

function ObservableLiteralSourceProxy:OnValueChanged(sender, e)
    self:RaiseValueChanged()
end

function ObservableLiteralSourceProxy:GetValue()
    return self.observableProperty.Value
end

function ObservableLiteralSourceProxy:Dispose()
    
end

return ObservableLiteralSourceProxy
