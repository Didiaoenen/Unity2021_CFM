local ComponentExt = require("LuaFramework.Views.Components.ComponentExt")

local MonoVariableArrayExt = class("MonoVariableArrayExt", ComponentExt)

MonoVariableArrayExt.CSObject = CS.Simple.MonoVariableArray

function MonoVariableArrayExt:ctor(handler)
   ComponentExt.ctor(self, handler)
end

function MonoVariableArrayExt:Get(name)
   local handler = self.handler
   return handler.variableArray:Get(name)
end

function MonoVariableArrayExt:dtor()
end

return MonoVariableArrayExt
