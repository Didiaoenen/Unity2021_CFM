local ComponentExt = require("LuaFramework.Views.Components.ComponentExt")

local TransformExt = class("TransformExt", ComponentExt)

TransformExt.CSObject = CS.UnityEngine.Transform

TransformExt.LocalRotation = "localRotation"

TransformExt.LocalPosition = "localPosition"

TransformExt.LocalScale = "localScale"

TransformExt.Parent = "parent"

function TransformExt.get:localRotation()
   local handler = self.handler
   return handler.localRotation
end

function TransformExt.set:localRotation(value)
   local handler = self.handler
   handler.localRotation = value
end

function TransformExt.get:localPosition()
   local handler = self.handler
   return handler.localPosition
end

function TransformExt.set:localPosition(value)
   local handler = self.handler
   handler.localPosition = value
end

function TransformExt.get:localScale()
   local handler = self.handler
   return handler.localScale
end

function TransformExt.set:localScale(value)
   local handler = self.handler
   handler.localScale = value
end

function TransformExt.set:parent(value)
   local handler = self.handler
   handler.parent = value
end

function TransformExt.get:parent()
   local handler = self.handler
   return handler.parent
end

function TransformExt:ctor(handler)
   ComponentExt.ctor(self, handler)
end

function TransformExt:SetParent(parent)
   local handler = self.handler
   handler:SetParent(parent.handler)
end

function TransformExt:dtor()
end

return TransformExt
