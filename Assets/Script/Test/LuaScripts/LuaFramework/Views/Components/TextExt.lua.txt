local ComponentExt = require("LuaFramework.Views.Components.ComponentExt")

local TextExt = class("TextExt", ComponentExt)

TextExt.CSObject = CS.UnityEngine.UI.Text

TextExt.Text = "text"

TextExt.Color = "color"

function TextExt.get:text()
   local handler = self.handler
   return handler.text
end

function TextExt.set:text(value)
   local handler = self.handler
   handler.text = value
end

function TextExt.get:color()
   local handler = self.handler
   return handler.color
end

function TextExt.set:color(value)
   local handler = self.handler
   handler.color = value
end

function TextExt.get:GameObject()
   return self.gameObject
end

function TextExt:ctor(handler)
   ComponentExt.ctor(self, handler)
end

function TextExt:dtor()
end

return TextExt
