local ComponentExt = require("LuaFramework.Views.Components.ComponentExt")

local ToggleExt = class("ToggleExt", ComponentExt)

ToggleExt.CSObject = CS.UnityEngine.UI.Toggle

ToggleExt.OnValueChanged = "onValueChanged"

function ToggleExt.get:onValueChanged()
   local handler = self.handler
   return handler.onValueChanged
end

function ToggleExt.set:onValueChanged(value)
   local handler = self.handler
   handler.onValueChanged = value
end

function ToggleExt.get:GameObject()
   return self.gameObject
end

function ToggleExt:ctor(handler)
   ComponentExt.ctor(self, handler)
end

function ToggleExt:dtor()
end

return ToggleExt
