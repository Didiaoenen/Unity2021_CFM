local TransformExt = require("LuaFramework.Views.Components.TransformExt")

local RectTransformExt = class("RectTransformExt", TransformExt)

RectTransformExt.CSObject = CS.UnityEngine.RectTransform

RectTransformExt.OffsetMax = "offsetMax"

RectTransformExt.OffsetMin = "offsetMin"

function RectTransformExt.get:offsetMax()
   local handler = self.handler
   return handler.offsetMax
end

function RectTransformExt.set:offsetMax(value)
   local handler = self.handler
   handler.offsetMax = value
end

function RectTransformExt.get:offsetMin()
   local handler = self.handler
   return handler.offsetMin
end

function RectTransformExt.set:offsetMin(value)
   local handler = self.handler
   handler.offsetMin = value
end

function RectTransformExt:ctor(handler)
   TransformExt.ctor(self, handler)
end

function RectTransformExt:dtor()
end

return RectTransformExt
