local ComponentExt = require("LuaFramework.Views.Components.ComponentExt")

local InputFieldExt = class("InputFieldExt", ComponentExt)

InputFieldExt.CSObject = CS.UnityEngine.UI.InputField

InputFieldExt.Text = "text"

InputFieldExt.OnEndEdit = "onEndEdit"

InputFieldExt.OnValueChanged = "onValueChanged"

function InputFieldExt.get:text()
   local handler = self.handler
   return handler.text
end

function InputFieldExt.set:text(value)
   local handler = self.handler
   handler.text = value
end

function InputFieldExt.get:onEndEdit()
   local handler = self.handler
   return handler.onEndEdit
end

function InputFieldExt.set:onEndEdit(value)
   local handler = self.handler
   handler.onEndEdit = value
end

function InputFieldExt.get:onValueChanged()
   local handler = self.handler
   return handler.onValueChanged
end

function InputFieldExt.set:onValueChanged(value)
   local handler = self.handler
   handler.onValueChanged = value
end

function InputFieldExt.get:GameObject()
   return self.gameObject
end

function InputFieldExt:ctor(handler)
   ComponentExt.ctor(self, handler)
end

function InputFieldExt:dtor()
end

return InputFieldExt
