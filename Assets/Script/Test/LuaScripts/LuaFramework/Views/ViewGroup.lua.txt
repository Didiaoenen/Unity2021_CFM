local ListExtention = require("LuaFramework.Common.ListExtention")

local ViewGroup = class("ViewGroup")

function ViewGroup:ctor()
   self.views = {}
end

function ViewGroup:OnEnable()
end

function ViewGroup:OnDisable()
end

function ViewGroup:OnShow()
   ListExtention.Foreach(self.views, function(_, v)
      v:DoShow()
   end)
end

function ViewGroup:OnHide()
   ListExtention.Foreach(self.views, function(_, v)
      v:DoHide()
   end)
   ListExtention.Foreach(self.views, function(_, v)
      v:DoDismiss()
   end)
end

function ViewGroup:GetView(name)
   assert({ name }, "")
end

function ViewGroup:AddView(name, parent)
   assert({ name, parent }, "")
end

function ViewGroup:RemoveView(name)
   assert({ name }, "")
end

function ViewGroup:ctor()
end

return ViewGroup
