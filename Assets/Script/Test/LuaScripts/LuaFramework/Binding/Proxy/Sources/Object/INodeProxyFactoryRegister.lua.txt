local INodeProxyFactoryRegister = class("INodeProxyFactoryRegister")

function INodeProxyFactoryRegister:ctor()
end

function INodeProxyFactoryRegister:Register(factory, priority)
   assert({ factory, priority }, "")
end

function INodeProxyFactoryRegister:Unregister(factory)
   assert({ factory }, "")
end

return INodeProxyFactoryRegister
