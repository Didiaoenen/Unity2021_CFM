local ISourceProxyFactoryRegistry = class("ISourceProxyFactoryRegistry")

function ISourceProxyFactoryRegistry:ctor()
end

function ISourceProxyFactoryRegistry:Register(factory, priority)
   assert({ factory, priority }, "")
end

function ISourceProxyFactoryRegistry:Unregister(factory)
   assert({ factory }, "")
end

return ISourceProxyFactoryRegistry
