local SourceProxyBase = require("LuaFramework.Binding.Proxy.Sources.SourceProxyBase")

local LiteralSourceProxy = class("LiteralSourceProxy", SourceProxyBase)

function LiteralSourceProxy:ctor(source)
    self.super:ctor(source)
end

function LiteralSourceProxy:GetValue()
    return self.source
end

return LiteralSourceProxy
