local BindingProxyBase = require("LuaFramework.Binding.Proxy.BindingProxyBase")
local ISourceProxy = require("LuaFramework.Binding.Proxy.Sources.ISourceProxy")

local SourceProxyBase = class("SourceProxyBase", BindingProxyBase, ISourceProxy)

function SourceProxyBase.get:Source()
   return self.source
end

function SourceProxyBase.set:Source(value)
   self.source = value
end

function SourceProxyBase:ctor(source)
   self.source = source
end

return SourceProxyBase
