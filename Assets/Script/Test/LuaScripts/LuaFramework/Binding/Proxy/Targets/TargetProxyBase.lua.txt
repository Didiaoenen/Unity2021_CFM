local BindingProxyBase = require("LuaFramework.Binding.Proxy.BindingProxyBase")
local ITargetProxy = require("LuaFramework.Binding.Proxy.Targets.ITargetProxy")
local BindingMode = require("LuaFramework.Binding.BindingMode")

local TargetProxyBase = class("TargetProxyBase", BindingProxyBase, ITargetProxy)

function TargetProxyBase.get:Target()
   return self.target
end

function TargetProxyBase.get:DefaultMode()
   return BindingMode.OneWay
end

function TargetProxyBase:ctor(target)
   BindingProxyBase.ctor(self)
   ITargetProxy.ctor(self)
   self.target = target
   self.targetName = ""
end

function TargetProxyBase:IsAlive(target)
   assert({ target })
   return true
end

function TargetProxyBase:dtor()
end

return TargetProxyBase
