local PathToken = class("PathToken")

function PathToken:ctor(paths, index)
   self.paths = paths
   self.index = index
end

function PathToken.get:Paths()
   return self.paths
end

function PathToken.get:Index()
   return self.index
end

function PathToken.get:Current()
   return self.paths[self.index]
end

function PathToken:HasNext()
   if #(self.paths) <= 0 or self.index >= #(self.paths) then
      return false
   end
   return true
end

function PathToken:NextToken()
   if not self:HasNext() then
      return
   end

   if nil == self.nextToken then
      self.nextToken = PathToken.new(self.paths, self.index + 1)
   end
   return self.nextToken
end

function PathToken:dtor()
end

return PathToken
