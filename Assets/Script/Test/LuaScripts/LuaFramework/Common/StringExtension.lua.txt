
function string.split(str, sep)
	sep = sep or ""
	local items = {}
    local pattern = string.format("([^%s]+)", sep)
    string.gsub(str, pattern, function (c)
        table.insert(items, c)
    end)  
    return items
end

function string.isnilorempty(str)
    if str == nil or str == "" then
        return true
    end
    return false
end
