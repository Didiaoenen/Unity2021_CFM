local ListExtention = {}

function ListExtention.Foreach(t, callback)
   for i, v in ipairs(t) do
      if callback then
         callback(i, v)
      end
   end
end

function ListExtention.FindIndex(t, obj)
   for i, v in ipairs(t) do
      if obj == v then
         return i
      end
   end
end

function ListExtention.Exists(t, obj)
   for _, v in ipairs(t) do
      if obj == v then
         return true
      end
   end
end

function ListExtention.Find(t, callback)
   for i, v in ipairs(t) do
      if callback and callback(i, v) then
         return v
      end
   end
end

function ListExtention.RemoveObj(t, obj)
   for i, v in ipairs(t) do
      if v == obj then
         return table.remove(t, i)
      end
   end
end

function ListExtention.Remove(t, callback)
   for i, v in ipairs(t) do
      if callback and callback(i, v) then
         return table.remove(t, i)
      end
   end
end

function ListExtention.ContainsValue(t, obj)
   for _, v in ipairs(t) do
      if v == obj then
         return true
      end
   end
end

return ListExtention
