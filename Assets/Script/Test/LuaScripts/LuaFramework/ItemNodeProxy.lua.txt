function ItemNodeProxy:ctor(source, key, itemInfo)
    self.key = key
    self.isList = source:GetName() == "List"
    self.itemInfo = itemInfo

    if self.source and self.source:GetName() == "INotifyCollectionChanged" then
        self.source.CollectionChanged = function ()
            self:OnCollectionChanged()
        end

    end
end

end

end

return ItemNodeProxy
