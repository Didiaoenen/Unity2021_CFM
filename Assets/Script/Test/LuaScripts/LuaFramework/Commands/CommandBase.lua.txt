local CommandBase = class("CommandBase")

function CommandBase.set:ExecuteChanged(value)
   setcallback(self.executeChanged, value)
end

function CommandBase:ctor()
   self.executeChanged = {}
end

function CommandBase:RaiseExecuteChanged()
   for _, v in ipairs(self.executeChanged) do
      v.callback(v.this)
   end
end

function CommandBase:CanExecute(...)
   assert({ ... }, "")
end

function CommandBase:Execute(...)
   assert({ ... }, "")
end

function CommandBase:dtor()
end

return CommandBase
