local CommandBase = class("CommandBase")

CommandBase.__setter.CanExecuteChanged = function (this, k, v)
    if v and v.callback ~= nil then
        table.insert(this.canExecuteChanged, v)
    else
        for i, value in ipairs(this.canExecuteChanged) do
            if v.this == value.this then
                table.remove(this.canExecuteChanged, i)
                break
            end
        end
    end
end

function CommandBase:ctor()
    self.canExecuteChanged = {}
end

function CommandBase:RaiseCanExecuteChanged()
    for i, v in ipairs(self.canExecuteChanged) do
        v.callback(v.this)
    end
end

function CommandBase:CanExecute(parameter)
    
end

function CommandBase:Execute(parameter)
    
end

return CommandBase
